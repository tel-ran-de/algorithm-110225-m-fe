function reverseBubbleSort(arr) {
    let n = arr.length;
    for (let i = 0; i < n - 1; i++) {
        for (let j = 0; j < n - i - 1; j++) {
            if (arr[j] < arr[j + 1]) {             // Изменено на '>' для сортировки в обратном порядке
                let temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }
    return arr;
}


Сложность алгоритма O(n^2)
Так как алгоритм использует два вложенных цикла для сравнения элементов массива.
